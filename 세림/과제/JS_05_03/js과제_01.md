    // 숫자(키) 입력
    let height = prompt("키를 입력해주세요. 공백으로 키를 구분합니다");
    // prompt에서 입력받은 값을 숫자로 치환해서 배열에 저장
    let numbers = height.split(" ").map(Number);

    // 저장된 값을 비교해서 오름차순으로 되어있는지 확인

    // 변수를 true로 초기화
    let vs = true;
    // 배열을 반복하면서 현재 숫자 > 다음숫자가 되면 false로 저장 후 for문을 빠져나옴
    for (let i = 0; i < numbers.length - 1; i++) {
        if (numbers[i] > numbers[i + 1]) {
        vs = false;
        break;
        }
    }
    // 위에 조건을 확인하고 오름차순이면 YES, 내림차순이면 NO출력
    if (vs) {
        document.write("입력 : " + height + "</br>");
        document.write("출력 : " + "YES");
        } else {
        document.write("입력 : " + height + "</br>");
        document.write("출력 : " + "NO");
    }

    // 이름 입력하기
    let name = prompt("반장선거투표");
    // 공백으로 저장
    let names = name.split(" ");
    // 입력된 값을 배열로 저장
    let a = {};
    // 횟수를 저장
    let count = 0;
    // 이름을 저장
    let naming = "";

    // 배열에서 가장 많이 나온 이름과 그 횟수를 찾기
    for(let i = 0; i < names.length; i++) {
        let nname = names[i];
        if(a[nname]) {
            a[nname]++;
        } else {
            a[nname] = 1;
        }

        if(a[nname] > count) {
            count = a[nname];
            naming = nname;
        }
    }
    // 도큐먼트에 결과 출력
    document.write(name + "</br>")
    document.write(naming + "(이)가 총" + count + "표로 반장이 되었습니다");
